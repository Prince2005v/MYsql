# 💻 Sample SQL Query

```sql
CREATE DATABASE company_data;

USE company_data;

CREATE TABLE employees (
    emp_id INT PRIMARY KEY,
    name VARCHAR(100),
    department VARCHAR(50),
    salary INT
);

INSERT INTO employees
VALUES
(101, 'Prince', 'Data Analyst', 45000),
(102, 'Rahul', 'HR', 35000),
(103, 'Aman', 'Developer', 60000);

SELECT department, AVG(salary) AS average_salary
FROM employees
GROUP BY department;
```
